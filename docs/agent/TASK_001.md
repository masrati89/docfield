### שלב 2 — Data Queries (React Query)

צור 3 queries. **חובה: organization_id על כל אחת.**

```typescript
// stats
const { data: stats } = useQuery({
  queryKey: ['home', 'stats', orgId],
  queryFn: async () => {
    const { data } = await supabase
      .from('delivery_reports')
      .select('status')
      .eq('organization_id', orgId);

    return {
      active:
        data?.filter((r) => ['draft', 'in_progress'].includes(r.status))
          .length ?? 0,
      completed:
        data?.filter((r) => ['completed', 'sent'].includes(r.status)).length ??
        0,
    };
  },
});

// recent reports (5 לפי updated_at)
const { data: recentReports } = useQuery({
  queryKey: ['home', 'reports', orgId],
  queryFn: async () => {
    const { data } = await supabase
      .from('delivery_reports')
      .select('id, title, status, updated_at, project_id')
      .eq('organization_id', orgId)
      .order('updated_at', { ascending: false })
      .limit(5);
    return data;
  },
});

// recent projects (3 לפי updated_at)
const { data: recentProjects } = useQuery({
  queryKey: ['home', 'projects', orgId],
  queryFn: async () => {
    const { data } = await supabase
      .from('projects')
      .select('id, name, address, type, updated_at')
      .eq('organization_id', orgId)
      .order('updated_at', { ascending: false })
      .limit(3);
    return data;
  },
});
```
